# Ame-no-Nuboko (Heavenly Jeweled Spear)
aom_ame_no_nuboko_template = {
	can_equip = {
		always = yes
	}
	can_benefit = {
		trigger_if = {
    		limit = { has_game_rule = aom_artifact_requirements_enabled }
			culture = { has_cultural_pillar = heritage_japanese }
		}
	}
	fallback = {
		prowess = 8
		martial = 2
	}
	unique = yes
}
