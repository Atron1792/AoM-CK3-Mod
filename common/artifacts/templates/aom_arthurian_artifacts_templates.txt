# Excalibur (Caledfwlch)
aom_excalibur_template = {
	can_equip = { always = yes }
	can_benefit = {
		trigger_if = {
    		limit = { has_game_rule = aom_artifact_requirements_enabled }
			culture = { has_cultural_pillar = heritage_brythonic }
        }
	}
	fallback = {
		prowess = 10
		martial = 2
		monthly_prestige = 0.5
	}
	unique = yes
}
