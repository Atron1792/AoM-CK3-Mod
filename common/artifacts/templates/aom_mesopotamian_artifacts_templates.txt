# Inventory Artifacts
############################
# Sharur (Ninurta's Mace)
aom_sharur_template = {
	can_equip = { always = yes }
	can_benefit = {
		trigger_if = {
			limit = { has_game_rule = aom_artifact_requirements_enabled }
			culture = {
				OR = { 
					has_cultural_pillar = heritage_caucasian
					has_cultural_pillar = heritage_syriac
				}
			}
		}
	}
	fallback = {
		prowess = 10
		stewardship = 2
		martial = 3
	}
	unique = yes
}
