# Sharur (Ninurta's Mace)
aom_sharur_template = {
	can_equip = { always = yes }
	can_benefit = {
		trigger_if = {
			limit = { has_game_rule = aom_artifact_requirements_enabled }
			culture = { has_cultural_pillar = heritage_mesopotamian }
		}
	}
	fallback = {
		prowess = 10
	}
	unique = yes
}
