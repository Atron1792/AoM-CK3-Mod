# Sleg (Spear of Lugh)
aom_sleg_template = {
	can_equip = {
		always = yes
	}
	can_benefit = {
		trigger_if = {
    		limit = { has_game_rule = aom_artifact_requirements_enabled }
			culture = {
				OR = { 
					has_cultural_pillar = heritage_brythonic
					has_cultural_pillar = heritage_goidelic
				}
			}
		}
	}
	fallback = {
		prowess = 10
		martial = 2
	}
	unique = yes
}
