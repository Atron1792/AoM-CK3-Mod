# Necronomicon
aom_necronomicon_template = {
	can_equip = { always = yes }
	can_benefit = {
		trigger_if = {
    		limit = { has_game_rule = aom_artifact_requirements_enabled }
			OR = {
				has_trait = lunatic
				has_trait = possessed
				has_trait = eccentric
			}
		}
	}
	fallback = {
		learning = 8
		stress_gain_mult = 0.5
	}
	unique = yes
}
