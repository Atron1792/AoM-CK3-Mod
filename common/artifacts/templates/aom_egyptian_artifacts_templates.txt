# Book of Thoth
aom_book_of_thoth_template = {
	can_equip = { always = yes }
	can_benefit = {
		trigger_if = {
			limit = { has_game_rule = aom_artifact_requirements_enabled }
			culture = { has_cultural_pillar = heritage_egyptian }
		}
	}
	fallback = {
        prowess = 4
    }
	unique = yes
}
