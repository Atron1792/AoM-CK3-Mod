# Book of Thoth
aom_book_of_thoth_template = {
	can_equip = { always = yes }
	can_benefit = {
		trigger_if = {
			limit = { has_game_rule = aom_artifact_requirements_enabled }
			OR = {
				culture = { has_cultural_pillar = heritage_east_african }
				root.culture = culture:egyptian 
			}
		}
	}
	fallback = {
        learning = 7
		monthly_learning_lifestyle_xp_gain_mult = 0.1
		clergy_opinion = 5
    }
	unique = yes
}
