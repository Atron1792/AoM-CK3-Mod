# Pangu’s Axe
aom_pangu_axe_template = {
	can_equip = {
		always = yes
	}
	can_benefit = {
		trigger_if = {
			limit = {
				has_game_rule = aom_artifact_requirements_enabled
			}
			culture = {
				OR = {
					has_cultural_pillar = heritage_chinese
					has_cultural_pillar = heritage_qiangic
				}
			}
		}
	}
	fallback = {
		prowess = 10
		monthly_prestige = 0.5
		court_grandeur_baseline_add = 5
	}
	unique = yes
}
