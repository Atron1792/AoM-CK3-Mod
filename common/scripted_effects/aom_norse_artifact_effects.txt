# Gungnir (Odin’s Spear)
create_artifact_aom_gungnir_effect = {
	$OWNER$ = {
		save_scope_as = owner
	}
	set_artifact_rarity_illustrious = yes
	create_artifact = {
		name = aom_gungnir_name
		description = aom_gungnir_desc
		type = weapon_simple
		visuals = aom_gungnir
		wealth = scope:wealth
		quality = scope:quality
		template = aom_gungnir_template
		history = {
			type = created_before_history
		}
		modifier = aom_gungnir_modifier
		save_scope_as = newly_created_artifact
		decaying = no
	}
	scope:newly_created_artifact = {
		set_variable = {
			name = historical_unique_artifact
			value = yes
		}
		set_variable = aom_gungnir
		set_variable = aom_artifact
		save_scope_as = epic
	}
}
