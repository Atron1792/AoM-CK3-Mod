# Inventory Artifacts
############################
# Sharur (Ninurta's Mace)
create_artifact_aom_sharur_effect = {
	$OWNER$ = { save_scope_as = owner }
	set_artifact_rarity_illustrious = yes
	create_artifact = {
		name = aom_sharur_name
		description = aom_sharur_desc
		type = weapon_simple
		visuals = aom_sharur
		wealth = scope:wealth
		quality = scope:quality
		template = aom_sharur_template
		history = { type = created_before_history }
		modifier = aom_sharur_modifier
		save_scope_as = newly_created_artifact
		decaying = no
	}
	scope:newly_created_artifact = {
		set_variable = { name = historical_unique_artifact value = yes }
		set_variable = aom_sharur
		set_variable = aom_artifact
		save_scope_as = epic
	}
}
