# Sleg (Spear of Lugh) 
create_artifact_aom_sleg_effect = {
	$OWNER$ = {
		save_scope_as = owner
	}
	set_artifact_rarity_illustrious = yes
	create_artifact = {
		name = aom_sleg_name
		description = aom_sleg_desc
		type = weapon_simple
		visuals = aom_sleg
		wealth = scope:wealth
		quality = scope:quality
		template = aom_sleg_template
		history = {
			type = created_before_history
		}
		modifier = aom_sleg_modifier
		save_scope_as = newly_created_artifact
		decaying = no
	}
	scope:newly_created_artifact = {
		set_variable = {
			name = historical_unique_artifact
			value = yes
		}
		set_variable = aom_sleg
		set_variable = aom_artifact
		save_scope_as = epic
	}
}
