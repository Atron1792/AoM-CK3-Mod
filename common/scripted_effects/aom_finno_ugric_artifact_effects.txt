# Inventory Artifacts
############################
# Ukonvasara (Hammer of Ukko)
create_artifact_aom_ukonvasara_effect = {
	$OWNER$ = { save_scope_as = owner }
	set_artifact_rarity_illustrious = yes
	create_artifact = {
		name = aom_ukonvasara_name
		description = aom_ukonvasara_desc
		type = weapon_simple
		visuals = aom_ukonvasara
		wealth = scope:wealth
		quality = scope:quality
		template = aom_ukonvasara_template
		history = { type = created_before_history }
		modifier = aom_ukonvasara_modifier
		save_scope_as = newly_created_artifact
		decaying = no
	}
	scope:newly_created_artifact = {
		set_variable = { name = historical_unique_artifact value = yes }
		set_variable = aom_ukonvasara
		set_variable = aom_artifact
		save_scope_as = epic
	}
}
