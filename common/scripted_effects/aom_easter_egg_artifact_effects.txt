############################
# Soul Edge (Soul Calibur)
create_artifact_aom_soul_edge_effect = {
	$OWNER$ = { save_scope_as = owner }
	set_artifact_rarity_illustrious = yes
	create_artifact = {
		name = aom_soul_edge_name
		description = aom_soul_edge_desc
		type = weapon_simple
		visuals = aom_soul_edge
		wealth = scope:wealth
		quality = scope:quality
		template = aom_soul_edge_template
		history = { type = created_before_history }
		modifier = aom_soul_edge_modifier
		save_scope_as = newly_created_artifact
		decaying = no
	}
	scope:newly_created_artifact = {
		set_variable = { name = historical_unique_artifact value = yes }
		set_variable = aom_soul_edge
		set_variable = aom_artifact
		save_scope_as = epic
	}
}
