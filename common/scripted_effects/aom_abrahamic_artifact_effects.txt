# Inventory Artifacts (Christian)
# Conquest's Crown (Horsemen of the Apocalypse)
create_artifact_aom_conquests_crown_effect = {
	$OWNER$ = { save_scope_as = owner }
	set_artifact_rarity_illustrious = yes
	create_artifact = {
		name = aom_conquests_crown_name
		description = aom_conquests_crown_desc
		type = helmet_simple
		visuals = aom_conquests_crown
		wealth = scope:wealth
		quality = scope:quality
		template = aom_conquests_crown_template
		history = {
			type = created_before_history
		}
		modifier = aom_conquests_crown_modifier
		save_scope_as = newly_created_artifact
		decaying = no
	}
	scope:newly_created_artifact = {
		set_variable = {
			name = historical_unique_artifact
			value = yes
		}
		set_variable = aom_conquests_crown
		set_variable = aom_artifact
		save_scope_as = epic
	}
}
