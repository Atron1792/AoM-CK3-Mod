# Excalibur (Caledfwlch)
create_artifact_aom_excalibur_effect = {
	$OWNER$ = {
		save_scope_as = owner
	}
	set_artifact_rarity_illustrious = yes
	create_artifact = {
		name = aom_excalibur_name
		description = aom_excalibur_desc
		type = weapon_simple
		visuals = aom_excalibur
		wealth = scope:wealth
		quality = scope:quality
		template = aom_excalibur_template
		history = {
			type = created_before_history
		}
		modifier = aom_excalibur_modifier
		save_scope_as = newly_created_artifact
		decaying = no
	}
	scope:newly_created_artifact = {
		set_variable = {
			name = historical_unique_artifact
			value = yes
		}
		set_variable = aom_excalibur
		set_variable = aom_artifact
		save_scope_as = epic
	}
}
