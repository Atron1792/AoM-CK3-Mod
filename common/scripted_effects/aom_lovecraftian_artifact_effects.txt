# Necronomicon 
create_artifact_aom_necronomicon_effect = {
	$OWNER$ = { save_scope_as = owner }
	set_artifact_rarity_illustrious = yes
	create_artifact = {
		name = aom_necronomicon_name
		description = aom_necronomicon_desc
		type = journal
		visuals = aom_necronomicon
		wealth = scope:wealth
		quality = scope:quality
		template = aom_necronomicon_template
		history = {
			type = created
			actor = character:1111000 # Abdul Alhazred
			recipient = character:1111000 # Abdul Alhazred
			date = 733.10.31
			location = province:5924 # Damascus 
		}
		modifier = aom_necronomicon_modifier
		save_scope_as = newly_created_artifact
		decaying = no
	}
	scope:newly_created_artifact = {
		set_variable = {
			name = historical_unique_artifact
			value = yes
		}
		set_variable = aom_necronomicon
		set_variable = aom_artifact
		save_scope_as = epic
	}
}
