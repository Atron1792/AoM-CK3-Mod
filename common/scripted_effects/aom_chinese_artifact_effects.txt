# Pangu’s Axe
create_artifact_aom_pangu_axe_effect = {
	$OWNER$ = { save_scope_as = owner }
	set_artifact_rarity_illustrious = yes
	create_artifact = {
		name = aom_pangu_axe_name
		description = aom_pangu_axe_desc
		type = weapon_simple
		visuals = aom_pangu_axe
		wealth = scope:wealth
		quality = scope:quality
		template = aom_pangu_axe_template
		history = { type = created_before_history }
		modifier = aom_pangu_axe_modifier
		save_scope_as = newly_created_artifact
		decaying = no
	}
	scope:newly_created_artifact = {
		set_variable = { name = historical_unique_artifact value = yes }
		set_variable = aom_pangu_axe
		set_variable = aom_chinese_artifact
		save_scope_as = epic
	}
}
