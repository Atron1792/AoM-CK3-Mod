# Book of Thoth
create_artifact_aom_book_of_thoth_effect = {
	$OWNER$ = { save_scope_as = owner }
	set_artifact_rarity_illustrious = yes
	create_artifact = {
		name = aom_book_of_thoth_name
		description = aom_book_of_thoth_desc
		type = journal
		visuals = aom_book_of_thoth
		wealth = scope:wealth
		quality = scope:quality
		template = aom_book_of_thoth_template
		history = { type = created_before_history }
		modifier = aom_book_of_thoth_modifier
		save_scope_as = newly_created_artifact
		decaying = no
	}
	scope:newly_created_artifact = {
		set_variable = { name = historical_unique_artifact value = yes }
		set_variable = aom_book_of_thoth
		set_variable = aom_artifact
		save_scope_as = epic
	}
}
