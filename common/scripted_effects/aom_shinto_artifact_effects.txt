# Ame-no-Nuboko (Heavenly Jeweled Spear)
create_artifact_aom_ame_no_nuboko_effect = {
    $OWNER$ = { save_scope_as = owner }
    set_artifact_rarity_illustrious = yes
    create_artifact = {
        name = aom_ame_no_nuboko_name
        description = aom_ame_no_nuboko_desc
        type = weapon_simple
        visuals = aom_ame_no_nuboko
        wealth = scope:wealth
        quality = scope:quality
        template = aom_ame_no_nuboko_template
        history = { type = created_before_history }
        modifier = aom_ame_no_nuboko_modifier
        save_scope_as = newly_created_artifact
        decaying = no
    }
    scope:newly_created_artifact = {
        set_variable = { name = historical_unique_artifact value = yes }
        set_variable = aom_ame_no_nuboko
        set_variable = aom_artifact
        save_scope_as = epic
    }
}
