# Achilles' Armor
create_artifact_aom_achilles_armor_effect = {
	$OWNER$ = { save_scope_as = owner }
	set_artifact_rarity_famed = yes
	create_artifact = {
		name = aom_achilles_armor_name
		description = aom_achilles_armor_desc
		type = armor_simple
		visuals = aom_achilles_armor
		wealth = scope:wealth
		quality = scope:quality
		template = aom_achilles_armor_template
		history = { type = created_before_history }
		modifier = aom_achilles_armor_modifier
		save_scope_as = newly_created_artifact
		decaying = no
	}
	scope:newly_created_artifact = {
		set_variable = {
			name = historical_unique_artifact
			value = yes
		}
		set_variable = aom_achilles_armor
		set_variable = aom_greco_roman_artifact
		save_scope_as = epic
	}
}
