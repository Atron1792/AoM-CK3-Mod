# Kladenets (Self-Swinging Sword)
create_artifact_aom_kladenets_effect = {
    $OWNER$ = { save_scope_as = owner }
    set_artifact_rarity_famed = yes
    create_artifact = {
        name = aom_kladenets_name
        description = aom_kladenets_desc
        type = weapon_simple
        visuals = aom_kladenets
        wealth = scope:wealth
        quality = scope:quality
        template = aom_kladenets_template
        history = { type = created_before_history }
        modifier = aom_kladenets_modifier
        save_scope_as = newly_created_artifact
        decaying = no
    }
    scope:newly_created_artifact = {
        set_variable = { name = historical_unique_artifact value = yes }
        set_variable = aom_kladenets
        set_variable = aom_artifact
        save_scope_as = epic
    }
}
