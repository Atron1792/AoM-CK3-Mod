# On Losing Egypt
yearly_global_pulse = { on_actions = { aom_on_egypt_title_lost } }

aom_on_egypt_title_lost = {
	trigger = {
		AND = {
            any_character_artifact = {
                OR = {
                    has_variable = aom_pschent
                }
		    }
            NOR = {
                has_title = title:k_egypt
                any_vassal = { has_title = title:k_egypt }
		    }
        }
	}
	effect = {
		trigger_event = aom_artifact_checks.0001
	}
}
