# Transfer All Artifacts
aom_debug_transfer_all_artifacts = {
	decision_group_type = debug
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_legend.dds"
	}
	desc = aom_debug_transfer_all_artifacts_desc
	is_shown = { debug_only = yes }
	is_valid_showing_failures_only = { is_available_adult_or_is_commanding = yes }
	ai_check_interval = 0
	effect = {
		every_living_character = {
			every_character_artifact = {
				limit = { 
					OR = {
						has_variable = aom_artifact
						has_variable = aom_norse_artifact
						has_variable = aom_greco_roman_artifact
						has_variable = aom_finno_ugric_artifact
						has_variable = aom_chinese_artifact
						has_variable = aom_celtic_artifact
						has_variable = aom_lovecraftian_artifact
						has_variable = aom_shinto_artifact
					}
				}
				set_owner = ROOT
			}
		}
	}
}
