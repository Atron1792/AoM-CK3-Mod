namespace = aom_game_rule

# Generate Norse Artifacts
aom_game_rule.001 = {
	scope = none
	hidden = yes
	immediate = {
		if = {
			limit = { has_game_rule = aom_generate_norse_artifacts_all }
			trigger_event = aom_norse_artifacts.001
			trigger_event = aom_norse_artifacts.002
		}
		if = {
			limit = { has_game_rule = aom_generate_norse_artifacts_only_god }
			trigger_event = aom_norse_artifacts.001
		}
		if = {
			limit = { has_game_rule = aom_generate_norse_artifacts_only_legend }
			trigger_event = aom_norse_artifacts.002
		}
	}
}
