aom_chinese_artifacts.001 = {
	scope = none
	hidden = yes
	immediate = {
		# Court Artifacts
		

		# Inventory Artifacts
		# Pangu’s Axe
        while = {
            count = 1
            random_ruler = {
                limit = {
                    OR = { 
                        has_government = tribal_government 
                        has_government = feudal_government 
                    }
                    NOT = { 
                        has_character_flag = aom_seeded_relic 
                        any_artifact = { has_variable = aom_pangu_axe } 
                    }
                    highest_held_title_tier >= tier_county
                    culture = {
				        OR = { 
					        has_cultural_pillar = heritage_chinese
					        has_cultural_pillar = heritage_qiangic
				        }
				    }
                }
                create_artifact_aom_pangu_axe_effect = { OWNER = this }
                add_character_flag = { 
                    flag = aom_seeded_relic 
                    days = 1 
                }
            }
        }
        # Ruyi Jingu Bang (Sun Wukong’s Staff)
        while = {
            count = 1
            random_ruler = {
                limit = {
                    OR = { 
                        has_government = tribal_government 
                        has_government = feudal_government 
                    }
                    NOT = { 
                        has_character_flag = aom_seeded_relic 
                        any_artifact = { has_variable = aom_ruyi_jingu_bang } 
                    }
                    highest_held_title_tier >= tier_county
					culture = {
				        OR = { 
					        has_cultural_pillar = heritage_chinese
					        has_cultural_pillar = heritage_qiangic
				        }
				    }
                }
                create_artifact_aom_ruyi_jingu_bang_effect = { OWNER = this }
                add_character_flag = { 
                    flag = aom_seeded_relic 
                    days = 1 
                }
            }
        }
	}
}
