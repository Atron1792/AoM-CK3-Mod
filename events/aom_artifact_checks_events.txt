# Lovecraftian Stress Gain
aom_artifact_checks.0002 = {
	type = character_event
	title = aom_artifact_checks.0002.t
	desc = aom_artifact_checks.0002.desc
	theme = medicine
	right_portrait = {
		character = root
		animation = stress
	}
	# normal reaction
	option = {
		name = aom_artifact_checks.0002.a
		add_stress = 40
		ai_chance = { base = 10 }
	}
	# eccentric reaction
	option = {
		name = aom_artifact_checks.0002.b
		add_stress = 20
		trigger = { has_trait = eccentric }
		trait = eccentric
		ai_chance = { base = 25 }
	}
	# lunatic reaction
	option = {
		name = aom_artifact_checks.0002.c
		add_stress = 10
		trigger = { has_trait = lunatic }
		trait = lunatic_1
		ai_chance = { base = 50 }
	}
	# possessed reaction
	option = {
		name = aom_artifact_checks.0002.d
		add_stress = -20
		trigger = { has_trait = possessed }
		trait = possessed_1
		ai_chance = { base = 100 }
	}
}
